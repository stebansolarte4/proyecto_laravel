CREATE DATABASE biblioteca_web;
USE biblioteca_web;
 
--  Tabla de Roles (Para definir permisos)
CREATE TABLE rol (
    id_rol INT AUTO_INCREMENT PRIMARY KEY,
    nombre_rol VARCHAR(50) NOT NULL UNIQUE
);
 
-- 2. Tabla de Categorías (Independiente)
CREATE TABLE categoria (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nombre_categoria VARCHAR(50) NOT NULL UNIQUE
);
 
-- 3. Tabla de Autores (Independiente)
CREATE TABLE autor (
    id_autor INT AUTO_INCREMENT PRIMARY KEY,
    nombre_autor VARCHAR(100) NOT NULL
);
 
-- 4. Tabla de Usuarios (Depende de rol)
CREATE TABLE usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    correo VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    fk_rol INT,
    FOREIGN KEY (fk_rol) REFERENCES rol(id_rol)
);
 
-- 5. Tabla de Administradores (Depende de rol)
CREATE TABLE administrador (
    id_admin INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    fk_rol INT,
    FOREIGN KEY (fk_rol) REFERENCES rol(id_rol)
);
 
-- 6. Tabla de Libros (Depende de categoria y autor)
CREATE TABLE libro (
    id_libro INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    fk_categoria INT,
    fk_autor INT,
    stock INT DEFAULT 1,
    FOREIGN KEY (fk_categoria) REFERENCES categoria(id_categoria) ON DELETE SET NULL,
    FOREIGN KEY (fk_autor) REFERENCES autor(id_autor) ON DELETE SET NULL
);

drop table libro;
 
-- 7. Tabla de Préstamos (Depende de usuario, libro y administrador)
CREATE TABLE prestamo (
    id_prestamo INT AUTO_INCREMENT PRIMARY KEY,
    fk_usuario INT,
    fk_libro INT,
    fecha_devolucion DATE NOT NULL,
    fecha_entrega DATETIME NULL,
    estado ENUM ('activo', 'devuelto', 'mora') DEFAULT 'activo',
    FOREIGN KEY (fk_usuario) REFERENCES usuario(id_usuario),
    FOREIGN KEY (fk_libro) REFERENCES libro(id_libro)
);
